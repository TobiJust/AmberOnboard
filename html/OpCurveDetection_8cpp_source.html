138 void OpCurveDetection::searchForMarking(Mat &img)
139 {
140  // temprary matrix to store the result
141  Mat foundMarkingOnly(Size(img.cols - 2, img.rows), img.type(), Scalar(0));
142 
143  // creating reference value to be able to adjust maximal marking width
144  int k_ref = lookupForFullImg[(int)(lookup[img.rows - 1] + 0.5)] - lookupForFullImg[(int)(lookup[img.rows - 1] + 0.5) + 1]; //k_ref - difference in pixels between 2meter and 3meter lines(used as reference value). k_n is defined below
145 
146  for (int i = 0; i < img.rows; i++)
147  {
148  //coefficient adjusting max marking width for current row of img
149  double k;
150  if (lookup[i] == max_distance) //special case for the last row
151  k = ((int)lookupForFullImg[lookup[i + 1]] - (int)lookupForFullImg[lookup[i]]) / k_ref;
152  else // case for all next rows
153  k = ((int)lookupForFullImg[lookup[i]] - (int)lookupForFullImg[lookup[i - 1]]) / k_ref;
154 
155  //last "black to white" jump
156  int last_b2w;
157  //maximal distance between "black to white" and following "white to black" jump
158  int max_dist_b2w_w2b = marking_width * k;
159  //prevent little value to affect detection
160  max_dist_b2w_w2b += (max_dist_b2w_w2b <= 3) ? 2 : 0;
161  //counter to count from last_b2w
162  int marking_width_counter = 0;
163 
164  //read next row
165  uchar* srcrow = img.ptr(i);
166  uchar* dstrow = foundMarkingOnly.ptr(i);
167 
168  // analyzing the row
169  for (int j = 1; j < img.cols - 1; j++)
170  {
171  //simple implementation of [1,0,-1] 3x1 mask/filter
172  int dif = (srcrow[j - 1] - srcrow[j + 1]);
173  if (dif > threshold) //if white to black jump found
174  {
175  if (marking_width_counter > 0) //if marking line width is good
176  {
177  //set a point in the middle of the marking
178  dstrow[(last_b2w + j) / 2] = 255;
179  //stop counter
180  marking_width_counter = 0;
181  }
182  }
183  else if (dif < -threshold) //if black to white jump found
184  {
185  //save position of this b2w jump
186  last_b2w = j;
187  //start counter
188  marking_width_counter = max_dist_b2w_w2b;
189  }
190  else // otherwise
191  //decrease counter
192  marking_width_counter--;
193  }
194  }
195  img = foundMarkingOnly;
196 }

486 OpCurveDetection::Polynomial OpCurveDetection::calcFunction(map<double, double> &points)
487 {
488  //Init values (E stays for Sum)
489  double Ex4 = 0.0;
490  double Ex3 = 0.0;
491  double Ex2 = 0.0;
492  double Ex = 0.0;
493  double n = points.size() * 1.0;
494  double Ex2y = 0.0;
495  double Exy = 0.0;
496  double Ey = 0.0;
497 
498  //init new polynom
499  Polynomial line;
500  line.pointsUsed = points.size();
501 
502  //use vertical and horizontal coordinates as x and y
503  map<double, double>::iterator it;
504  for (it = points.begin(); it != points.end(); it++)
505  {
506  double x = it->first;
507  double y = it->second;
508  double x2 = x * x; //x^2
509  Ey += y;
510  Exy += x * y;
511  Ex2y += x2 * y;
512  Ex += x;
513  Ex2 += x2;
514  Ex3 += x2 * x;
515  Ex4 += x2 * x2;
516  }
517 
518  //Matrices to calculate an approximation polynomial a*x^2 + b*x + c = 0
519  // A B C
520  // [Ex^4 Ex^3 Ex^2] [a] [Ex^2y]
521  // [Ex^3 Ex^2 Ex ] x [b] = [Exy ]
522  // [Ex^2 Ex n ] [c] [Ey ]
523  Mat A = (Mat_<double>(3, 3, CV_64FC1) << Ex4, Ex3, Ex2, Ex3, Ex2, Ex, Ex2, Ex, n);
524  Mat B = Mat_<double>(3, 1, CV_64FC1);
525  Mat C = (Mat_<double>(3, 1, CV_64FC1) << Ex2y, Exy, Ey);
526 
527  //if equation is solvable
528  if (cv::determinant(A) != 0)
529  {
530  B = A.inv() * C;
531  line.a = B.at<double>(0, 0);
532  line.b = B.at<double>(1, 0);
533  line.c = B.at<double>(2, 0);
534  line.ERROR_FLAG = false;
535  }else
536  {
537  line.ERROR_FLAG = true; //if the equation has no solutions
538  }
539 
540  return line;
541 }
551 double OpCurveDetection::getRadius(Polynomial &l, Polynomial &r)
552 {
553  double radius;
554  //polynomial is good if it contains at least 1/4 of maximum points count (rows clount)
555  //left line has more points in it
556  if (l.pointsUsed > r.pointsUsed && l.pointsUsed > nlookup/4)
557  {
558  radius = calcRadius(l);
559  }//right line has more points in it
560  else if (l.pointsUsed < r.pointsUsed && r.pointsUsed > nlookup/4)
561  {
562  radius = calcRadius(r);
563  }//left line has same points count as right
564  else if (l.pointsUsed == r.pointsUsed && r.pointsUsed > nlookup/4)
565  {
566  radius = 0.5 * calcRadius(l) + 0.5 * calcRadius(r);
567  }//both lines have less that 1/4 of max points count
568  else
569  {
570  radius = 0; //radius calculation is not reliable
571  }
572  return radius;
573 }
