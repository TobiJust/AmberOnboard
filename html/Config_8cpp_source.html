361 bool Config::shrink(string &target, string compare, size_t start, bool substr) {
362 
363  // Check if index is in string bounds.
364  if ( !target.length() || !compare.length() )
365  return false;
366 
367  // Initialize lower and upper bounds.
368  size_t lower=start, upper=start;
369 
370  // Find last occurrence before start.
371  if ((lower=target.find_last_not_of(compare,lower)) == string::npos)
372  lower=0;
373  else if (target.find_last_not_of(compare.c_str(), lower, 1) != string::npos)
374  lower++;
375 
376  // Find first occurrence after start.
377  // Since erase ignores upper bound, no index condition must be handled.
378  if ((upper=target.find_first_not_of(compare,upper))==string::npos)
379  upper=target.size();
380 
381  // Erase or create concerned substring.
382  if (upper > lower) {
383  if (substr)
384  target=target.substr(lower, upper-lower);
385  else
386  target.erase(target.begin()+lower, target.begin()+upper);
387  }
388 
389  // Operation successful.
390  return true;
391 }

578

581  this->errors.push_back(toString(source, CONF_ERR_TUPLE));
582  return;
583  }
584 
585  // Convert key to lower case for comparison.
586  transform(tmp[0].begin(), tmp[0].end(), tmp[0].begin(), ::tolower);
587 
588  if (this->parsed.find(tmp[0]) != this->parsed.end())
589  status=CONF_ERR_ALREADY_SET;
590 
591  else {
592

594  if (EQUALS(tmp[0], 0, OPT_OBU_ID))

607

610  status = procIface(tmp, this->deferred);
611

613  else if (EQUALS(tmp[0], 0, OPT_CAP_OUT))
614  status = procCapture(tmp, this->outer);
615 
616  // Extract index of inner camera.
617  else if (EQUALS(tmp[0], 0, OPT_CAP_IN))
618  status = procCapture(tmp, this->inner);
619 
620  // Extract index of primary camera.
621  else if (EQUALS(tmp[0], 0, OPT_CAP_PRIME))
622  status = procPrimary(tmp, this->capPrimary);
623 
624  // Extract compression rate.

627 
628  // Extract gps port data.
629  else if (EQUALS(tmp[0], 0, OPT_GPS_DEV))

631

635 
636  // Extract acc port data.
637  else if (EQUALS(tmp[0], 0, OPT_ACC_DEV))

640  // Extract acc device type.

644  // Extract gps port data.

646  status = procOBD(tmp, this->obd);

648  else // Not a valid option.
649  status = CONF_ERR_UNKNOWN_OPT;
650  }
651

654  else this->parsed.insert(tmp[0]);
655 }
