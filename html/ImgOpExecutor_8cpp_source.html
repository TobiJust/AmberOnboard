410 
411  // At least primary image source exists.
412  if (this->imageCaptures.size()) {
413 
414  this->captureMutex.lock();
415

417  shared_ptr<cv::Mat> newest(this->imageCaptures[0]->getFrame());
418  shared_ptr<cv::Mat> temp;
419 
420  // 'newest' does not point to NULL.
421  if (newest) {
422 
423  // Get iterator for operator instances.
424  auto opIt = this->imageOperators.begin();
425 
426  // Iterate list of execution.
427  while (opIt != this->imageOperators.end()) {
428 
429  stringstream cap;
430  cap << ARG_CAPTURE << (int)0;
431  (*opIt)->setValue(cap.str(), shared_ptr<ValMat>(new ValMat(newest)));
432 
433  // If operator requires more than one capture.
434  for (uint8_t capture=1;

436  capture++) {
437

441

444  temp = this->imageCaptures[capture]->getFrame();
445  shared_ptr<ValMat> argValue(new ValMat(temp));
446  (*opIt)->setValue(cap.str(), argValue);
447  }
448 
449  // Process image from primary capture.

451  (*opIt)->apply(instanceResults);
