109 uint8_t ModuleNetworking::pollMsgFromChildren() {
110 
111  // Return value.
112  uint8_t result = 0;
113 
114  // Iterator for communicators.
115  auto commIt = this->communicators.begin();
116 
117  // Poll pending input fields from each communicator.
118  while (commIt != this->communicators.end()) {
119 
120  // Do while there are pending messages from child.
121  while ((*commIt)->in_count()) {
122 
123  // Get next message.
124  shared_ptr<Message_M2C> next = (*commIt)->in_pop();
125 
126  // Switch incoming message type.
127  switch (next->getType()) {
128  case MSG_DATA_ACQUIRED:
129  {
130  // Attach communicator to message type.
131  this->attachChildToMsg(*commIt, MSG_DATA_COMPLETE);
132 
133  // Send new data acquisition to message hub.
134  shared_ptr<M2M_DataAcquired> acquire(new M2M_DataAcquired());
135  MsgHub::getInstance()->appendMsg(acquire);
136 
137  break;
138  }
139  case MSG_COMMAND:
140  {
141  // A command was send, so get command type
142  shared_ptr<ValInt> cmdType(new ValInt);
143  shared_ptr<Value> tmp = dynamic_pointer_cast<Value>(cmdType);
144  string typeString = ARG_COMMAND_TYPE;
145  next->getValue(typeString, tmp);
146 
147  // Send new command to message hub.
148  shared_ptr<M2M_Command> command(new M2M_Command);
149  command->setValue(typeString, cmdType);
150  MsgHub::getInstance()->appendMsg(command);
151 
152  break;
153  }
154  case MSG_RESPAWN:
155  {
156  // The communicator requests a respawn.
157  shared_ptr<M2M_Respawn> respawn(new M2M_Respawn);
158  shared_ptr<ValInt> respID(new ValInt);
159 
160  // Set its identifier for command message.
161  switch ((*commIt)->getCommType()) {
162  case NW_TYPE_REALTIME:
163  respID->setValue(NW_REALTIME);
164  break;
165  case NW_TYPE_DEFERRED:
166  respID->setValue(NW_DEFERRED);
167  break;
168  default:
169  break;
170  }
171 
172  // Communicator could be deleted since it terminated.
173  this->com_delete(*commIt);
174 
175  // Send new respawn request to message hub.
176  respawn->setValue(ARG_RESPAWN_CHILD, respID);
177  MsgHub::getInstance()->appendMsg(respawn);
178 
179  return result;
180  }
181 
182  default:
183  break;
184  }
185  }
186  commIt++;
187  }
