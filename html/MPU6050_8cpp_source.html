55 int MPU6050::run() {
56 
57  // Run until terminate is called.
58  while (this->bus->isActive() && !this->isTerminating()) {
59 
60  int16_t tmp_ax, tmp_ay, tmp_az;
61  int16_t tmp_gx, tmp_gy, tmp_gz;
62 
63  getMotion6(&tmp_ax, &tmp_ay, &tmp_az, &tmp_gx, &tmp_gy, &tmp_gz);
64 
65  this->rw_mutex.lock();
66  this->addValue(this->values, ACC_X, to_string(tmp_ax));
67  this->addValue(this->values, ACC_Y, to_string(tmp_ay));
68  this->addValue(this->values, ACC_Z, to_string(tmp_az));
69  this->addValue(this->values, GYRO_X, to_string(tmp_gx));
70  this->addValue(this->values, GYRO_Y, to_string(tmp_gy));
71  this->addValue(this->values, GYRO_Z, to_string(tmp_gz));
72  this->rw_mutex.unlock();
73 
74  usleep(10000);
75 
76  }
77 
78  return 0;
79 }

347 void MPU6050::getMotion6(int16_t* ax, int16_t* ay, int16_t* az, int16_t* gx, int16_t* gy, int16_t* gz) {
348  this->bus->readBytes(this->devAddr, MPU6050_RA_ACCEL_XOUT_H, 14, this->buffer);
349  *ax = (((int16_t)buffer[0]) << 8) | buffer[1];
350  *ay = (((int16_t)buffer[2]) << 8) | buffer[3];
351  *az = (((int16_t)buffer[4]) << 8) | buffer[5];
352  *gx = (((int16_t)buffer[8]) << 8) | buffer[9];
353  *gy = (((int16_t)buffer[10]) << 8) | buffer[11];
354  *gz = (((int16_t)buffer[12]) << 8) | buffer[13];
355 }
