465 shared_ptr<Message_M2M> Initializer::processMsg(shared_ptr<Message_M2M> msg) {
466 
467  // Resulting message.
468  shared_ptr<Message_M2M> result;
469 
470  // Process message depending on message type.
471  switch (msg->getType()) {
472  case MSG_RESPAWN:
473  {
474  // Get type of child thread to respawn.
475  shared_ptr<Value> typeVal;
476  msg->getValue(ARG_RESPAWN_CHILD, typeVal);
477 
478  // Do requested respawn.
479  switch (dynamic_pointer_cast<ValInt>(typeVal)->getValue()) {
480 
481  // Respawn realtime communicator.
482  case NW_REALTIME: {
483 
484  // Get configuration of realtime connection.
485  server realtime;
486  if (!conf->getRealTime(realtime)) {
487  printErr(INIT_ERR_LOAD, "server params");
488  return msg;
489  }
490 
491  // Get device id.
492  uint8_t devID=0;
493  conf->getDeviceID(devID);
494 
495  // Create network communicator instance.
496  shared_ptr<NetworkCommunicator> comm = createComm(realtime.target, realtime.port, realtime.iface, NW_TYPE_REALTIME, devID);
497 
498  // If comm does not point to null, communicator creation was successful.
499  if (comm) {
500 
501  // Append network communicator to network module.
502  if (this->nw.com_append(comm)){
503  printErr(INIT_ERR_DEV_APPEND, "communicator");
504  return msg;
505  }
506 
507  // Add child to list of threaded children and run it.
508  this->nw.addChild(comm);
509  comm->attachObserver(&(this->nw));
510  this->nw.runChild(this->nw.getChildCount()-1);
511 
512  } else return msg;
513  break;
514  }
515 
516  // Respawn deferred communicator.
517  case NW_DEFERRED:
518  break;
519  default:
520  break;
521  }
522  break;
523  }
524  case MSG_TERM_BROADCAST:
525  {
526  // Terminate is requested.
527  this->terminate();
528  break;
529  }
530  default:
531  break;
532  }
533 
534  return result;
535 }
