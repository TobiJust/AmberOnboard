127 
128  if ( lock ) this->busMutex.lock();
129 
130  // Bus already open.
131  if (this->isOpen()) {
132  if (force)
133  this->release(false);

135  return IO_OK;

147  return IO_ERR_OPEN;
148  }
149 
150  if ( lock ) this->busMutex.unlock();
151 
152  this->setActive(true);
153 
154  return IO_OK;

279 uint8_t IOi2cBus::readBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t *data) {
280  // 01101001 read byte
281  // 76543210 bit numbers
282  // xxx args: bitStart=4, length=3
283  // 010 masked
284  // -> 010 shifted
285  uint8_t status, b;
286  if ( !(status = readByte(devAddr, regAddr, &b)) ) {

312 uint8_t IOi2cBus::readBytes(uint8_t devAddr, uint8_t regAddr, uint8_t length, uint8_t *data) {
313 
314  this->busMutex.lock();
315 
316  uint8_t status=IO_OK;
317 
318  // Open bus if necessary.
319  if ( !this->isOpen() )
320  status = this->request();
321 
322  // An error occurred.
323  if ( status ) {
324  this->busMutex.unlock();
325  return status;
326  }
327 
328  // Select i2c slave address.
329  if ( ioctl(this->fileDesc, I2C_SLAVE, devAddr) < 0 ) {
330  this->busMutex.unlock();
331  return IO_ERR_SETUP;
332  }
333 
334  // Write requested register to i2c bus.
335  if ( ::write(this->fileDesc, &regAddr, 1) != 1 ) {
336  this->busMutex.unlock();
337  return IO_ERR_WRITE;
338  }
339 
340  // Read data from i2c bus.
341  if ( ::read(this->fileDesc, data, length) != length )
342  status=IO_ERR_READ;
343 
344  this->busMutex.unlock();
345 
346  return status;
347 }

371 uint8_t IOi2cBus::writeBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t data) {
372  // 010 value to write
373  // 76543210 bit numbers
374  // xxx args: bitStart=4, length=3
375  // 00011100 mask byte
376  // 10101111 original value (sample)
377  // 10100011 original & ~mask
378  // 10101011 masked | value
379  uint8_t b;
380  if ( !readByte(devAddr, regAddr, &b) ) {
381  uint8_t mask = ((1 << length) - 1) << (bitStart - length + 1);
382  data <<= (bitStart - length + 1); // shift data into correct position
383  data &= mask; // zero all non-important bits in data
384  b &= ~(mask); // zero all important bits in existing byte
385  b |= data; // combine data with existing byte
386  return writeByte(devAddr, regAddr, b);
387  } else {
388  return IO_ERR_READ;
389  }
390 }

495  this->busMutex.unlock();

497  }
